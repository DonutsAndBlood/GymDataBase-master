relatorio 1: agrupar alunos por tipo de grupo muscular sendo treinado
relatorio 2: relatorio de exercicios relacionados com o alunos (quais exercicios ele vai fazer num dia)
relatorio 3: quant de alunos cadastrados
relatorio 4: quant de exercicios cadastrados
relatorio 5: quant de alunos com mensalidade paga

**rel 1**:
SELECT A.NOME_ALUNO, E.GRUPO_MUSCULAR, COUNT(*) AS ALUNOS
FROM ALUNOS A
INNER JOIN EXERCICIOS E ON E.CODIGO_EXERCICIO = A.CODIGO_EXERCICIO
GROUP BY A.NOME_ALUNO, E.GRUPO_MUSCULAR;

**rel 2**
SELECT  E.GRUPO_MUSCULAR, E.NOME_EXERCICIO, E.REPETICOES FROM EXERCICIOS E
INNER JOIN ALUNOS A ON E.CODIGO_EXERCICIO = A.CODIGO_EXERCICIO
WHERE CODIGO_ALUNO = :CODIGO_ALUNO
AND ROWNUM < 8
ORDER BY DBMS_RANDOM.VALUE

**rel 3**
SELECT COUNT(*) AS ALUNOS CADASTRADOS FROM ALUNOS

**rel 4**
SELECT COUNT(*) AS EXERCICIOS CADASTRADOS FROM EXERCICIOS

**rel 5**
SELECT  NOME_ALUNOS, COUNT(*) AS PAGOS FROM ALUNOS
WHERE PAGAMENTO = "P"
